        let currentQuestion = 0;

        let score = 0;

        let playerName = "";

        let startTime, timerInterval;


        const startScreen = document.getElementById('start-screen');

        const quizScreen = document.getElementById('quiz-screen');

        const resultsScreen = document.getElementById('results-screen');

        const leaderboardBody = document.getElementById('leaderboard-body');


        function startGame() {

            playerName = document.getElementById('student-name-input').value.trim();

            if (playerName === "") {

                alert("探險家，請先輸入你嘅姓名啊！");

                return;

            }

            document.getElementById('player-name-display').textContent = playerName;

            startScreen.classList.remove('active');

            quizScreen.classList.add('active');

            startTimer();

            loadQuestion();

        }


        function loadQuestion() {

            if (currentQuestion >= questionsData.length) {

                endQuiz();

                return;

            }

            const data = questionsData[currentQuestion];

            document.getElementById('quiz-image').src = data.image;

            document.getElementById('question-text').textContent = data.question;

            const optionsContainer = document.getElementById('options-container');

            optionsContainer.innerHTML = "";

            data.options.forEach((option, index) => {

                const button = document.createElement('button');

                button.textContent = option;

                button.className = 'option-btn';

                button.onclick = () => selectAnswer(index);

                optionsContainer.appendChild(button);

            });

        }


        function selectAnswer(selectedIndex) {

            if (selectedIndex === questionsData[currentQuestion].answer) {

                score++;

                document.getElementById('score-display').textContent = score;

            }

            currentQuestion++;

            loadQuestion();

        }


        function startTimer() {

            startTime = Date.now();

            timerInterval = setInterval(() => {

                document.getElementById('timer-display').textContent = Math.floor((Date.now() - startTime) / 1000);

            }, 1000);

        }


        function endQuiz() {

            clearInterval(timerInterval);

            const finalElapsedTime = Math.floor((Date.now() - startTime) / 1000);

            quizScreen.classList.remove('active');

            resultsScreen.classList.add('active');

            document.getElementById('final-score').textContent = score;

            document.getElementById('final-time').textContent = finalElapsedTime;

            

            updateLeaderboard(playerName, score, finalElapsedTime);

        }


        function resetGame() {

            currentQuestion = 0;

            score = 0;

            document.getElementById('score-display').textContent = "0";

            resultsScreen.classList.remove('active');

            startScreen.classList.add('active');

            document.getElementById('student-name-input').value = "";

        }


        // 雲端排行榜連線功能

        function updateLeaderboard(name, finalScore, time) {

            leaderboardBody.innerHTML = "<tr><td colspan='4'>正在努力將成績傳送到太空基地... 🚀<br>(請稍候幾秒鐘)</td></tr>";

            if (GOOGLE_SCRIPT_URL === "https://script.google.com/macros/s/AKfycbzI5jhgR8a_tVwr_Lf2AIveJ5cqKPRaobtbAGCJqQr4CsAWpdW775cdlnvgXmr9_XjVgg/exec") {

                leaderboardBody.innerHTML = "<tr><td colspan='4' style='color:red;'>⚠️ 老師請注意：你尚未在程式碼中貼上 Google Apps Script 的網址！</td></tr>";

                return;

            }


            // 組合要傳送的網址與資料 (使用 GET 請求傳遞參數)

            const urlWithData = `${GOOGLE_SCRIPT_URL}?action=save&name=${encodeURIComponent(name)}&score=${finalScore}&time=${time}`;


            // 發送請求到 Google Sheets

            fetch(urlWithData)

                .then(response => response.json())

                .then(data => {

                    leaderboardBody.innerHTML = "";

                    const medals = ["🥇 冠軍", "🥈 亞軍", "🥉 季軍"];

                    

                    if (data.length === 0) {

                        leaderboardBody.innerHTML = "<tr><td colspan='4'>暫時未有探險家完成任務！</td></tr>";

                        return;

                    }


                    data.forEach((entry, index) => {

                        const row = `<tr>

                            <td>${medals[index] || (index + 1)}</td>

                            <td>${entry.name}</td>

                            <td>${entry.score}</td>

                            <td>${entry.time}s</td>

                        </tr>`;

                        leaderboardBody.innerHTML += row;

                    });

                })

                    console.error('Error:', error);

                    leaderboardBody.innerHTML = "<tr><td colspan='4'>哎呀！太空通訊中斷，無法載入排行榜。📡</td></tr>";

                });

        }
